focus <- "Antarctica"
lon0 <- 0
lat0 <- -90
proj <- str_glue("+proj=ortho lon_0={lon0} lat_0={lat0}")
Code 
world <- st_read("data/world.gpkg") %>%
  st_make_valid() %>%
  select(CONTINENT) %>%
  st_transform(proj) %>%
  st_make_valid() %>%
  filter(!st_is_empty(.)) %>%
  group_by(CONTINENT) %>%
  summarise() 

continent <- world %>%
  filter(CONTINENT == focus)

graticule <- get_graticule(spacing = 15)

tmap

tm_shape(world) +
  tm_fill() + 
  tm_shape(graticule) +
  tm_lines(col = "lightgrey", lwd = 0.5) +
  tm_compass() +
  tm_layout(frame = FALSE)
